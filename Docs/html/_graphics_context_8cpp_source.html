136 bool Initialize(const int width, const int height, const char* title) {
137 g_InitWindowWidth = width;
138 g_InitWindowHeight = height;
139 g_WindowTitle = title;
140
141 if (glfwInit() != GLFW_TRUE) {
142 return false;
143 }
144 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 4);
145 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 6);
146 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
147 glfwWindowHint(GLFW_SAMPLES, 4);
148
149#ifndef NDEBUG
150 glfwWindowHint(GLFW_OPENGL_DEBUG_CONTEXT, true);
151#endif
152
153 g_Window = GL::TWindow(glfwCreateWindow(width, height, g_WindowTitle, nullptr, nullptr));
154 if (g_Window == nullptr) {
155 glfwTerminate();
156 return false;
157 }
158 glfwMakeContextCurrent(g_Window.get());
159
160 if (!gladLoadGLLoader(reinterpret_cast<GLADloadproc>(glfwGetProcAddress))) {
161 return false;
162 }
163
164 GL::FramebufferCallback(g_Window.get(), width, height);
165 glfwSetFramebufferSizeCallback(GetWindow(), GL::FramebufferCallback);
166 glfwSwapInterval(0);
167 glfwSetInputMode(g_Window.get(), GLFW_CURSOR, GLFW_CURSOR_DISABLED);
168
169 Logger::LogInfo(Logger::Subsystems::GRAPHICS, "Graphics subsystem initialized.");
170
171 return true;
172 }
