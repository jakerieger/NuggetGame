156 bool Initialize(const char* title) {
157 const auto width = static_cast<i32>(Settings::GetSettings().Graphics.ResolutionX);
158 const auto height = static_cast<i32>(Settings::GetSettings().Graphics.ResolutionY);
159 g_InitWindowWidth = width;
160 g_InitWindowHeight = height;
161 g_WindowTitle = title;
162
163 if (glfwInit() != GLFW_TRUE) {
164 return false;
165 }
166 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 4);
167 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 6);
168 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
169 glfwWindowHint(GLFW_SAMPLES, 4);
170
171#ifndef NDEBUG
172 glfwWindowHint(GLFW_OPENGL_DEBUG_CONTEXT, true);
173#endif
174
175 g_Window = GL::TWindow(glfwCreateWindow(width, height, g_WindowTitle, nullptr, nullptr));
176 if (g_Window == nullptr) {
177 glfwTerminate();
178 return false;
179 }
180 glfwMakeContextCurrent(g_Window.get());
181
182 if (!gladLoadGLLoader(reinterpret_cast<GLADloadproc>(glfwGetProcAddress))) {
183 return false;
184 }
185
186 GL::FramebufferCallback(g_Window.get(), width, height);
187 glfwSetFramebufferSizeCallback(GetWindow(), GL::FramebufferCallback);
188
189 glfwSwapInterval(Settings::GetSettings().Graphics.Vsync ? 1 : 0);
190
191 // glfwSetInputMode(g_Window.get(), GLFW_CURSOR, GLFW_CURSOR_DISABLED);
192
193 Logger::LogInfo(Logger::Subsystems::GRAPHICS, "Graphics subsystem initialized.");
194
195 return true;
196 }
