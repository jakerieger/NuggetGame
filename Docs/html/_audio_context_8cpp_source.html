20 static bool CheckALErrors() {
21 if (const ALenum error = alGetError(); error != AL_NO_ERROR) {
22 switch (error) {
23 case AL_INVALID_NAME:
24 Logger::LogError(
25 Logger::Subsystems::AUDIO,
26 "AL_INVALID_NAME: a bad name (ID) was passed to an OpenAL function");
27 break;
28 case AL_INVALID_ENUM:
29 Logger::LogError(
30 Logger::Subsystems::AUDIO,
31 "AL_INVALID_ENUM: an invalid enum value was passed to an OpenAL function");
32 break;
33 case AL_INVALID_VALUE:
34 Logger::LogError(
35 Logger::Subsystems::AUDIO,
36 "AL_INVALID_VALUE: an invalid value was passed to an OpenAL function");
37 break;
38 case AL_INVALID_OPERATION:
39 Logger::LogError(Logger::Subsystems::AUDIO,
40 "AL_INVALID_OPERATION: the requested operation is not valid");
41 break;
42 case AL_OUT_OF_MEMORY:
43 Logger::LogError(Logger::Subsystems::AUDIO,
44 "AL_OUT_OF_MEMORY: the requested operation resulted in OpenAL "
45 "running out of memory");
46 break;
47 default:
48 Logger::LogError(Logger::Subsystems::AUDIO, "UNKNOWN AL ERROR: %s", error);
49 }
50 return false;
51 }
52 return true;
53 }

92 void Initialize() {
93 const auto deviceName = alcGetString(nullptr, ALC_DEFAULT_DEVICE_SPECIFIER);
94
95 g_Device = alcOpenDevice(deviceName);
96 if (!g_Device) {
97 Logger::LogError(Logger::Subsystems::AUDIO, "Could not open AL device.");
98 }
99
100 g_Context = alcCreateContext(g_Device, nullptr);
101 g_ContextCurrent = alcMakeContextCurrent(g_Context);
102 if (!g_ContextCurrent) {
103 Logger::LogError(Logger::Subsystems::AUDIO, "Failed to make AL context current.");
104 }
105
106 CheckALCErrors(g_Device);
107 CheckALErrors();
108
109 Logger::LogInfo(Logger::Subsystems::AUDIO, "Audio subsystem initialized.");
110 }

112 static std::tuple<u32, u32>
113 PlaySoundFile(const std::string& filename, const bool loop = false, const f32 gain = 1.f) {
114 AudioFile<f32> oneShot;
115 oneShot.shouldLogErrorsToConsole(false);
116 if (!oneShot.load(filename)) {
117 Logger::LogError(Logger::Subsystems::AUDIO,
118 "Failed to load audio file: '%s'",
119 filename.c_str());
120 }
121
122 ALenum format;
123 if (oneShot.isMono()) {
124 format = AL_FORMAT_MONO_FLOAT32;
125 } else {
126 format = AL_FORMAT_STEREO_FLOAT32;
127 }
128
129 u32 alSource;
130 u32 alSampleSet;
131 std::vector<f32> samples;
132 samples.reserve(oneShot.getNumSamplesPerChannel() * oneShot.getNumChannels());
133
134 if (oneShot.isMono()) {
135 samples = oneShot.samples.at(0);
136 } else {
137 samples = Utilities::InterleaveVectors(oneShot.samples.at(0), oneShot.samples.at(1));
138 }
139
140 // OpenAL requires that the number of samples be a multiple of the data type size in bytes
141 // (float in this case) * the number of channels. If it's not, this line adds additional
142 // silence at the end of the buffer to make it a multiple.
143 samples.resize(
144 Utilities::MakeMultiple(static_cast<i32>(samples.size()),
145 static_cast<i32>(sizeof(f32)) * oneShot.getNumChannels()),
146 0.f);
147
148 alGenSources(1, &alSource);
149 alGenBuffers(1, &alSampleSet);
150 alBufferData(alSampleSet,
151 format,
152 samples.data(),
153 static_cast<ALsizei>(samples.size()),
154 static_cast<ALsizei>(oneShot.getSampleRate()));
155 alSourcei(alSource, AL_BUFFER, static_cast<ALint>(alSampleSet));
156 alSourcei(alSource, AL_LOOPING, loop);
157 alSourcef(alSource, AL_GAIN, gain);
158
159 oneShot.samples.clear();
160
161 alSourcePlay(alSource);
162
163 return std::make_tuple(alSource, alSampleSet);
164 }

166 static void Cleanup(const u32 source, const u32 buffer) {
167 ALint sourceState;
168 do {
169 alGetSourcei(source, AL_SOURCE_STATE, &sourceState);
170 } while (sourceState == AL_PLAYING);
171 alDeleteSources(1, &source);
172 alDeleteBuffers(1, &buffer);
173 }

175 void PlayOneShot(const std::string& filename, const f32 gain) {
176 const std::tuple<u32, u32> result = PlaySoundFile(filename, false, gain);
177 const auto source = std::get<0>(result);
178 const auto buffer = std::get<1>(result);
179
180 auto cleanupThread = std::thread(Cleanup, source, buffer);
181 cleanupThread.detach();
182 }
