66 void Update(IGameApp& app) {
67 Graphics::ResetDrawCalls();
68 Graphics::UpdateFrameTime();
69 Camera::Update();
70
71 const float frameTime = Graphics::GetDeltaTime();
72 const auto activeScene = app.GetActiveScene();
73
74 // Update scene
75 if (activeScene) {

77 activeScene->Update(frameTime);

80// Update engine analytics

86 // Clear buffers

98 if (activeScene) {
99 activeScene->Render();
100 }

103 UI::Draw();
104 UI::End();

106#ifndef NDEBUG

128 void RunApp(IGameApp& app) {
129 double t = 0.0;
130 constexpr double dt = FIXED_TIMESTEP / ADJUSTMENT;
131 double currentTime = glfwGetTime();
132 double accumulator = 0.0;
133
134 while (RUNNING) {
135 const double newTime = glfwGetTime();
136 const double frameTime = newTime - currentTime;
137 currentTime = newTime;
138 accumulator += frameTime;
139 while (accumulator >= dt) {
140 FixedUpdate(app);
141 accumulator -= dt;
142 t += dt;
143 }
144
145 Update(app);
146 }
147
148 app.Cleanup();
149 Shutdown();
150 }

157void IGameApp::LoadScene(const std::string& name) {
158 const auto scene = GetScene(name);
159 if (const auto currentScene = GetActiveScene()) {
160 UnloadScene(currentScene);
161 }
162
163 if (scene) {
164 scene->SetActive(true);
165 scene->Start();
166
167 for (auto& go : scene->GetContext().GameObjects) {
168 Input::RegisterListener(go.get());
169 }
170 }
