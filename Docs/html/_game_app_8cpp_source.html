144 void RunApp(IGameApp& app) {
145 double t = 0.0;
146 constexpr double dt = FIXED_TIMESTEP / ADJUSTMENT;
147 double currentTime = glfwGetTime();
148 double accumulator = 0.0;
149
150 while (RUNNING) {
151 const double newTime = glfwGetTime();
152 const double frameTime = newTime - currentTime;
153 currentTime = newTime;
154 accumulator += frameTime;
155 while (accumulator >= dt) {
156 if (!g_IsLoading) {
157 FixedUpdate(app);
158 }
159
160 accumulator -= dt;
161 t += dt;
162 }
163
164 if (!g_IsLoading) {
165 Update(app);
166 }
167 }
168
169 app.Cleanup();
170 Shutdown();
171 }
