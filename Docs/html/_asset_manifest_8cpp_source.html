15 static bool ValidateManifest(const Document& manifest) {
16 if (!manifest.IsObject())
17 return false;
18 if (!manifest.HasMember("version") || !manifest["version"].IsNumber())
19 return false;
20 if (!manifest.HasMember("assets") || !manifest["assets"].IsArray())
21 return false;
22
23 return true;
24 }

26 void AssetManifest::ParseManifest(const Document& manifest) {
27 m_Version = manifest["version"].GetInt();
28 m_Filename = manifest["filename"].GetString();
29 const auto assets = manifest["assets"].GetArray();
30 if (assets.Size() > 0) {
31 for (const auto& asset : assets) {
32 const auto name = asset["name"].GetString();
33 const auto type = asset["type"].GetString();
34 const auto properties = asset["properties"].GetObject();
35
36 if (strcmp(type, "sprite") == 0) {
37 assert(manifest.HasMember("filename"));
38 const auto filename = asset["filename"].GetString();
39 const auto descriptor = new SpriteDescriptor;
40 descriptor->m_Name = name;
41 descriptor->m_Version = m_Version;
42 auto srcBytes = IO::ReadAllBytes(filename);
43 if (!srcBytes.has_value())
44 break;
45 descriptor->m_SrcData = srcBytes.value();
46 descriptor->GetProperties()->m_Height = properties["height"].GetInt();
47 descriptor->GetProperties()->m_Width = properties["width"].GetInt();
48 descriptor->GetProperties()->m_IsAlpha = properties["alpha"].GetBool();
49
50 m_Descriptors.push_back(descriptor);
51 }
52
53 if (strcmp(type, "font") == 0) {
54 assert(manifest.HasMember("filename"));
55 const auto filename = asset["filename"].GetString();
56 const auto descriptor = new FontDescriptor;
57 descriptor->m_Name = name;
58 descriptor->m_Version = m_Version;
59 auto srcBytes = IO::ReadAllBytes(filename);
60 if (!srcBytes.has_value())
61 break;
62 descriptor->m_SrcData = srcBytes.value();
63 descriptor->GetProperties()->m_DefaultSize = properties["defaultSize"].GetInt();
64
65 m_Descriptors.push_back(descriptor);
66 }
67
68 if (strcmp(type, "audio") == 0) {
69 assert(manifest.HasMember("filename"));
70 const auto filename = asset["filename"].GetString();
71 const auto descriptor = new AudioDescriptor;
72 descriptor->m_Name = name;
73 descriptor->m_Version = m_Version;
74 auto srcBytes = IO::ReadAllBytes(filename);
75 if (!srcBytes.has_value())
76 break;
77 descriptor->m_SrcData = srcBytes.value();
78 descriptor->GetProperties()->m_SampleCount = properties["sampleCount"].GetInt();
79 descriptor->GetProperties()->m_SampleRate = properties["sampleRate"].GetInt();
80 descriptor->GetProperties()->m_Channels = properties["channels"].GetInt();
81
82 m_Descriptors.push_back(descriptor);
83 }
84
85 if (strcmp(type, "level") == 0) {
86 const auto descriptor = new LevelDescriptor;
87 descriptor->m_Name = name;
88 descriptor->m_Version = m_Version;
89 auto srcBytes = asset["data"].GetArray();
90 std::vector<u8> data;
91 std::vector<i32> srcData;
92 for (int i = 0; i < srcBytes.Size(); i++) {
93 auto value = srcBytes[i].GetInt();
94 srcData.push_back(value);
95 }
96 data.resize(sizeof(i32) * srcData.size());
97 std::copy(reinterpret_cast<u8*>(srcData.data()),
98 reinterpret_cast<u8*>(srcData.data() + srcData.size()),
99 data.begin());
100 descriptor->m_SrcData = data;
101
102 descriptor->GetProperties()->m_Rows = properties["rows"].GetInt();
103 descriptor->GetProperties()->m_Columns = properties["columns"].GetInt();
104 descriptor->GetProperties()->m_PlayerStart = {
105 properties["playerStart"]["x"].GetFloat(),
106 properties["playerStart"]["y"].GetFloat(),
107 };
108 descriptor->GetProperties()->m_ObjectivePosition = {
109 properties["objectivePosition"]["x"].GetFloat(),
110 properties["objectivePosition"]["y"].GetFloat(),
111 };
112
113 m_Descriptors.push_back(descriptor);
114 }
115 }
116 }

119 AssetManifest::AssetManifest(const std::filesystem::path& manifest) {
120 const auto readResult = IO::Read(manifest);
121 if (!readResult.has_value()) {
122 throw std::runtime_error("Failed to read manifest file");
123 }
124
125 const std::string& manifestSource = readResult.value();
126 Document manifestDocument;
127 manifestDocument.Parse(manifestSource.c_str());

131 }

146 std::optional<std::vector<u8>> AssetManifest::Serialize() {
147 std::vector<u8> bytes;
148
149 int reserveSize = std::accumulate(m_Descriptors.begin(),
150 m_Descriptors.end(),
151 0,
152 [](int acc, const IAssetDescriptor* descriptor) {
153 return acc + (int)descriptor->GetSize();
154 });
155 const u32 assetCount = m_Descriptors.size();
156 reserveSize += sizeof(u32);
157 bytes.resize((size_t)reserveSize);
158
159 memcpy(bytes.data(), &assetCount, sizeof(u32));
160 size_t offset = sizeof(u32);
161
162 for (const auto& descriptor : m_Descriptors) {
163 auto descriptorBytes = descriptor->Serialize();
164 const auto offsetDelta = descriptor->GetSize();
165 memcpy(bytes.data() + offset, descriptorBytes.data(), offsetDelta);
166 offset += offsetDelta;
167 }
168
169 return bytes;
170 }
