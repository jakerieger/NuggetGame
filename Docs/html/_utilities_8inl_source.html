39 inline u32 RGBAToHex(const f32 r, const f32 g, const f32 b, const f32 a) {
40 const auto redByte = static_cast<unsigned char>(r * 255.0f);
41 const auto greenByte = static_cast<unsigned char>(g * 255.0f);
42 const auto blueByte = static_cast<unsigned char>(b * 255.0f);
43 const auto alphaByte = static_cast<unsigned char>(a * 255.0f);
44
45 return (alphaByte << 24) | (redByte << 16) | (greenByte << 8) | blueByte;
46 }

201 std::vector<T> InterleaveVectors(const std::vector<T>& vec1, const std::vector<T>& vec2) {
202 std::vector<T> result;
203 const size_t size = std::min(vec1.size(), vec2.size());
204
205 for (size_t i = 0; i < size; ++i) {
206 result.push_back(vec1[i]);
207 result.push_back(vec2[i]);
208 }
209
210 // Add any remaining elements from the longer vector
211 for (size_t i = size; i < vec1.size(); ++i) {
212 result.push_back(vec1[i]);
213 }
214 for (size_t i = size; i < vec2.size(); ++i) {
215 result.push_back(vec2[i]);
216 }
217
218 return result;
219 }
