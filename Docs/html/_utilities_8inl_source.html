40 inline u32 RGBAToHex(const f32 r, const f32 g, const f32 b, const f32 a) {
41 const auto redByte = static_cast<unsigned char>(r * 255.0f);
42 const auto greenByte = static_cast<unsigned char>(g * 255.0f);
43 const auto blueByte = static_cast<unsigned char>(b * 255.0f);
44 const auto alphaByte = static_cast<unsigned char>(a * 255.0f);
45
46 return (alphaByte << 24) | (redByte << 16) | (greenByte << 8) | blueByte;
47 }

89 inline u32 LoadTextureFromFile(const char* path) {
90 u32 id;
91 glGenTextures(1, &id);
92
93 i32 width, height, nrComponents;
94 stbi_set_flip_vertically_on_load(true);
95 u8* data = stbi_load(path, &width, &height, &nrComponents, 0);
96 if (data) {
97 GLenum format = GL_RGB;
98 if (nrComponents == 1) {
99 format = GL_RED;
100 } else if (nrComponents == 3) {
101 format = GL_RGB;
102 } else if (nrComponents == 4) {
103 format = GL_RGBA;
104 }
105
106 glBindTexture(GL_TEXTURE_2D, id);
107 glTexImage2D(GL_TEXTURE_2D,
108 0,
109 static_cast<int>(format),
110 width,
111 height,
112 0,
113 format,
114 GL_UNSIGNED_BYTE,
115 data);
116 glGenerateMipmap(GL_TEXTURE_2D);
117
118 glTexParameteri(GL_TEXTURE_2D,
119 GL_TEXTURE_WRAP_S,
120 format == GL_RGBA
121 ? GL_CLAMP_TO_EDGE
122 : GL_REPEAT); // for this tutorial: use GL_CLAMP_TO_EDGE to
123 // prevent semi-transparent borders. Due to
124 // interpolation it takes texels from next repeat
125 glTexParameteri(GL_TEXTURE_2D,
126 GL_TEXTURE_WRAP_T,
127 format == GL_RGBA ? GL_CLAMP_TO_EDGE : GL_REPEAT);
128 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
129 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
130
131 stbi_image_free(data);
132 } else {
133 stbi_image_free(data);
134 }
135
136 return id;
137 }

211 std::vector<T> InterleaveVectors(const std::vector<T>& vec1, const std::vector<T>& vec2) {
212 std::vector<T> result;
213 const size_t size = std::min(vec1.size(), vec2.size());
214
215 for (size_t i = 0; i < size; ++i) {
216 result.push_back(vec1[i]);
217 result.push_back(vec2[i]);
218 }
219
220 // Add any remaining elements from the longer vector
221 for (size_t i = size; i < vec1.size(); ++i) {
222 result.push_back(vec1[i]);
223 }
224 for (size_t i = size; i < vec2.size(); ++i) {
225 result.push_back(vec2[i]);
226 }
227
228 return result;
