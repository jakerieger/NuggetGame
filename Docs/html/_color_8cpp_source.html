23f32 AColor::Luminance() const {
24 // See <https://www.w3.org/TR/WCAG20/#relativeluminancedef>
25 const auto uRed = static_cast<u32>(Red * 255);
26 const auto uGreen = static_cast<u32>(Green * 255);
27 const auto uBlue = static_cast<u32>(Blue * 255);
28
29 const auto R = LinearizeComponent(static_cast<f32>(uRed) / 0xFF);
30 const auto G = LinearizeComponent(static_cast<f32>(uGreen) / 0xFF);
31 const auto B = LinearizeComponent(static_cast<f32>(uBlue) / 0xFF);
32
33 return 0.2126f * R + 0.7152f * G + 0.0722f * B;
34}

65AColor AColor::AlphaBlend(const AColor& foreground, const AColor& background) {
66 const f32 alpha = foreground.Alpha;
67 if (alpha == 0.f) {
68 return background;
69 }
70 const f32 invAlpha = 1.f - alpha;
71
72 f32 backAlpha = background.Alpha;
73 if (backAlpha == 1.f) {
74 return {alpha * foreground.Red + invAlpha * background.Red,
75 alpha * foreground.Green + invAlpha * background.Green,
76 alpha * foreground.Blue + invAlpha * background.Blue,
77 1.f};
78 }
79
80 backAlpha = (backAlpha * invAlpha) / 1.f;
81 const f32 outAlpha = alpha + backAlpha;
82 assert(outAlpha != 0.f);
83 return {alpha * foreground.Red + invAlpha * background.Red / outAlpha,
84 alpha * foreground.Green + invAlpha * background.Green / outAlpha,
85 alpha * foreground.Blue + invAlpha * background.Blue / outAlpha,
86 1.f};
87}
