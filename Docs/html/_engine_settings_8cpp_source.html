29 bool ReadSettings() {
30 const mINI::INIFile config(Resources::GetConfigFile().string());
31 mINI::INIStructure ini;
32 if (!config.read(ini)) {
33 Logger::LogError(Logger::Subsystems::RUNTIME,
34 "Unable to read config.ini. File missing or corrupt.");
35 return false;
36 }
37
38 g_Settings.Graphics.Vsync = ini["Graphics"]["Vsync"] == "true";
39 g_Settings.Graphics.Fullscreen = ini["Graphics"]["Fullscreen"] == "true";
40 g_Settings.Graphics.ResolutionX = std::stoi(ini["Graphics"]["ResolutionX"]);
41 g_Settings.Graphics.ResolutionY = std::stoi(ini["Graphics"]["ResolutionY"]);
42
43 g_Settings.Audio.VolumeMaster = std::stof(ini["Audio"]["VolumeMaster"]);
44 g_Settings.Audio.VolumeUI = std::stof(ini["Audio"]["VolumeUI"]);
45 g_Settings.Audio.VolumeFX = std::stof(ini["Audio"]["VolumeFX"]);
46 g_Settings.Audio.VolumeMusic = std::stof(ini["Audio"]["VolumeMusic"]);
47
48 Logger::LogInfo(Logger::Subsystems::RUNTIME, "Loaded game settings from config.ini");
49 return true;
50 }
