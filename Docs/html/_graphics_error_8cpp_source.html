12 void EnableDebugOutput() {
13 int flags;
14 glGetIntegerv(GL_CONTEXT_FLAGS, &flags);
15 if (flags & GL_CONTEXT_FLAG_DEBUG_BIT) {
16 Logger::LogInfo(Logger::Subsystems::GRAPHICS, "Graphics debugging is enabled");
17 glEnable(GL_DEBUG_OUTPUT);
18 glEnable(GL_DEBUG_OUTPUT_SYNCHRONOUS);

22 Logger::LogError(Logger::Subsystems::GRAPHICS,

24 }
