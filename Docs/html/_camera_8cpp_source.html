38 glm::vec2 ScreenToWorld(const glm::vec2& screenPoint) {
39 const f32 u = screenPoint.x / g_Width;
40 const f32 v = (g_Height - screenPoint.y) / g_Height;
41
42 const f32 ratio = g_Width / g_Height;
43 glm::vec2 extents(ratio * 25.0f, 25.0f);
44 extents *= g_Zoom;
45
46 const glm::vec2 lower = g_Center - extents;
47 const glm::vec2 upper = g_Center + extents;
48
49 glm::vec2 pw;
50 pw.x = (1.0f - u) * lower.x + u * upper.x;
51 pw.y = (1.0f - v) * lower.y + v * upper.y;
52 return pw;
53 }

55 glm::vec2 WorldToScreen(const glm::vec2& worldPoint) {
56 const f32 ratio = g_Width / g_Height;
57 glm::vec2 extents(ratio * 25.0f, 25.0f);
58 extents *= g_Zoom;
59
60 const glm::vec2 lower = g_Center - extents;
61 const glm::vec2 upper = g_Center + extents;
62
63 const float u = (worldPoint.x - lower.x) / (upper.x - lower.x);
64 const float v = (worldPoint.y - lower.y) / (upper.y - lower.y);
65
66 glm::vec2 ps;
67 ps.x = u * g_Width;
68 ps.y = (1.0f - v) * g_Height;
69 return ps;
70 }

72 glm::mat4 BuildProjectionMatrix(f32 zBias) {
73 const f32 w = g_Width;
74 const f32 h = g_Height;
75 const float ratio = w / h;
76 glm::vec2 extents(ratio * 25.0f, 25.0f);
77 extents *= g_Zoom;
78
79 const glm::vec2 lower = g_Center - extents;
80 const glm::vec2 upper = g_Center + extents;
81
82 // X Axis vector
83 const glm::vec4 x = {2.0f / (upper.x - lower.x), 0.f, 0.f, 0.f};
84 // Y Axis vector
85 const glm::vec4 y = {0.f, 2.0f / (upper.y - lower.y), 0.f, 0.f};
86 // Z Axis Vector
87 constexpr glm::vec4 z = {0.f, 0.f, 1.0f, 0.f};
88 // Zoom(??) vector
89 const glm::vec4 zm = {-(upper.x + lower.x) / (upper.x - lower.x),
90 -(upper.y + lower.y) / (upper.y - lower.y),
91 zBias,
92 1.f};

95 m[0] = x;
96 m[1] = y;
97 m[2] = z;

106 glm::mat4 BuildViewMatrix() {
107 auto view = glm::mat4(1.f);
108
109 f32 targetX = 0.f;
110 f32 targetY = 0.f;
111 if (g_FollowTarget) {
112 const auto targetPos = g_FollowTarget->GetTransform()->GetPosition();
113 if (g_PreviousTargetPosition == glm::vec2(0.f)) {
114 g_PreviousTargetPosition = targetPos;
115 }
116
117 const auto lerpedX = g_PreviousTargetPosition.x +
118 k_LerpFactor * (targetPos.x - g_PreviousTargetPosition.x);
119 g_PreviousTargetPosition.x = lerpedX;
120 targetX = lerpedX;
121
122 const auto lerpedY = g_PreviousTargetPosition.y +
123 (k_LerpFactor / 3.f) * (targetPos.y - g_PreviousTargetPosition.y);
124 g_PreviousTargetPosition.y = lerpedY;
125 targetY = lerpedY + k_OffsetY;
126 }
127 // TODO: Handle Y following differently from simply lerping to target Y
128 view = glm::translate(view, {-targetX, -targetY + 10.f, 0.f});
129
130 return view;
131 }
