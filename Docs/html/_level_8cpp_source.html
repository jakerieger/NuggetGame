33 // Invert tile id map to match coordinate system in game

36 // Generate x,y positions for each tile
37 auto y = 0;
38 for (const auto& row : m_TileIds) {
39 y++;
40 auto x = 0;
41 for (auto id : row) {
42 x++;
43 auto xPos = static_cast<f32>(x) * k_TileSize * 2.f - 2.f;
44 auto yPos = static_cast<f32>(y) * k_TileSize * 2.f - 2.f;
45 // 0 is the empty tile, we can ignore it to save memory
46 if (id != 0) {
47 m_TilePositions[id].emplace_back(xPos, yPos);
48
49 // Create the colliders
50 b2BodyDef bodyDef;
51 bodyDef.position.Set(xPos, yPos + 0.5f);
52 const auto body = Physics::GetWorld()->CreateBody(&bodyDef);
53 b2PolygonShape bodyShape;
54 bodyShape.SetAsBox(2.f, 2.f);
55 body->CreateFixture(&bodyShape, 0.f);
56 }
57 }
58 }

60 // Create a tilemap for each tile
61 for (auto& [id, positions] : m_TilePositions) {
