40 std::optional<PackError> PackSprites(const std::filesystem::path& spritesDir,
41 const std::filesystem::path& dataDir) {
42 std::vector<Schemas::Sprite> sprites;
43
44 // For each sprite asset in our Assets directory
45 {
46 std::ranges::for_each(
47 fs::directory_iterator {spritesDir},
48 [&](const fs::directory_entry& dirEntry) {
49 const auto entry = dirEntry.path().string();
50 int width, height, nrComponents;
51 stbi_set_flip_vertically_on_load(true);
52 unsigned char* data = stbi_load(entry.c_str(), &width, &height, &nrComponents, 0);
53 if (data) {
54 const auto filename = dirEntry.path().filename().string();
55 sprites.emplace_back(filename.c_str(), width, height, nrComponents, data);
56 }
57 stbi_image_free(data);
58 });
59 }
60
61 // Compress raw bytes and write metadata
62 std::vector<char> rawBytes;
63 char* compressedData;
64 int compressedDataSize;
65 {
66 for (const auto& sprite : sprites) {
67 // Copy width, height, and name
68 rawBytes.insert(rawBytes.end(),
69 reinterpret_cast<const char*>(&sprite.nameLen),
70 reinterpret_cast<const char*>(&sprite.nameLen) +
71 sizeof(sprite.nameLen));
72 rawBytes.insert(rawBytes.end(),
73 reinterpret_cast<const char*>(&sprite.dataLen),
74 reinterpret_cast<const char*>(&sprite.dataLen) +
75 sizeof(sprite.dataLen));
76 rawBytes.insert(rawBytes.end(),
77 reinterpret_cast<const char*>(&sprite.width),
78 reinterpret_cast<const char*>(&sprite.width) +
79 sizeof(sprite.width));
80 rawBytes.insert(rawBytes.end(),
81 reinterpret_cast<const char*>(&sprite.height),
82 reinterpret_cast<const char*>(&sprite.height) +
83 sizeof(sprite.height));
84 rawBytes.insert(rawBytes.end(),
85 reinterpret_cast<const char*>(&sprite.channels),
86 reinterpret_cast<const char*>(&sprite.channels) +
87 sizeof(sprite.channels));
88 rawBytes.insert(rawBytes.end(),
89 reinterpret_cast<const char*>(sprite.name),
90 reinterpret_cast<const char*>(sprite.name) +
91 MAX_STR_LEN); // Include null terminator
92
93 // Copy image data
94 rawBytes.insert(rawBytes.end(),
95 sprite.data,
96 sprite.data +
97 sprite.width * sprite.height * 4); // Assuming RGBA format
98 }
99
100 const size_t srcSize = sizeof(char) * rawBytes.size();
101
102 // Write original for debugging
103 // {
104 // std::ofstream origFile(dataDir / "sprites_orig.nugpak", std::ios::binary);
105 // origFile.write(rawBytes.data(), srcSize);
106 // origFile.close();
107 // }
108
109 const size_t maxDstSize = LZ4_compressBound(static_cast<int>(srcSize));
110 compressedData = static_cast<char*>(malloc(maxDstSize));
111 compressedDataSize = LZ4_compress_default(rawBytes.data(),
112 compressedData,
113 static_cast<int>(srcSize),
114 static_cast<int>(maxDstSize));
115 if (compressedDataSize <= 0) {
116 return PackError::ERROR_COMPRESSION_FAILED;
117 }
118
119 // ReSharper disable once CppDFAConstantConditions
120 if (compressedDataSize > 0) {
121 printf("Done. Ratio: %.2f\n",
122 static_cast<float>(compressedDataSize) / static_cast<float>(srcSize));
123 }
124
125 compressedData =
126 static_cast<char*>(realloc(compressedData, static_cast<size_t>(compressedDataSize)));
127 if (!compressedData) {
128 return PackError::ERROR_REALLOC_FAILED;
129 }
130
131 // Write NUGMETA file
132 auto outFile = dataDir / "sprites.nugmeta";
133 std::ofstream metaFile(outFile, std::ios::binary);
134 if (!metaFile.is_open()) {
135 return PackError::ERROR_OPEN_FILE_FOR_WRITE_FAILED;
136 }
137
138 // NUGMETA binary layout:
139 // (size_t) (int) (uint) size_t[]
140 // 00 00 00 00 00 00 00 00 | 00 00 00 00 | 00 00 00 00 | [00 00 00 00 00 00 00 00]...
141 // Origin size Comp size Num elements Block sizes
142 auto numElements = static_cast<unsigned int>(sprites.size());
143 metaFile.write(reinterpret_cast<const char*>(&srcSize), sizeof(size_t));
144 metaFile.write(reinterpret_cast<const char*>(&compressedDataSize), sizeof(int));
145 metaFile.write(reinterpret_cast<const char*>(&numElements), sizeof(unsigned int));
146
147 std::vector<char> blockSizeBytes;
148 for (auto i = 0; i < sprites.size(); i++) {
149 auto sprite = sprites.at(i);
150 size_t blockSize = 0;
151 blockSize += sizeof(unsigned int) * 5;
152 blockSize += sprite.dataLen;
153 blockSize += MAX_STR_LEN;
154
155 blockSizeBytes.insert(blockSizeBytes.end(),
156 reinterpret_cast<const char*>(&blockSize),
157 reinterpret_cast<const char*>(&blockSize) + sizeof(size_t));
158 }
159
160 metaFile.write(blockSizeBytes.data(), sizeof(size_t) * sprites.size());
161 metaFile.close();
162 }
163
164 // Write compressed data to file
165 {
166 auto outFile = dataDir / "sprites.nugpak";
167 std::ofstream pakFile(outFile, std::ios::binary);
168 if (!pakFile.is_open())
169 return PackError::ERROR_OPEN_FILE_FOR_WRITE_FAILED;
170
171 pakFile.write(compressedData, compressedDataSize);
172 pakFile.close();
173 }
174
175 // Cleanup
176 {
177 for (const auto& sprite : sprites) {
178 delete[] sprite.data;
179 delete[] sprite.name;
180 }
181
182 free(compressedData);
183 }
184
185 return {};
186 }
