49 static std::string GetTimestamp(time_t rawtime) {
50 char buffer[80];
51 time(&rawtime);
52 const tm* timeinfo = localtime(&rawtime);
53 strftime(buffer, 80, "%m/%d/%Y @ %I:%M:%S %p", timeinfo);
54 auto timestamp = std::string(buffer);

56 return timestamp;

73 static std::string FormatEntryString(const FLogEntry& entry) {
74 const char* logLevel = g_LevelMap.find(entry.Level)->second;
75 const char* subsystem = entry.Subsystem.c_str();

85 const auto buffer = new char[bufferSize + 1];
86 buffer[bufferSize] = '\0';
87 sprintf(buffer, "|%s| [%s] (%s)\n => %s", timestamp.c_str(), logLevel, subsystem, message);
88 std::string out = STRDUP(buffer);

93 static void Log(const FLogEntry& entry) {
94 g_LogEntries.push_back(entry);
95
96 const char* logLevel = g_LevelMap.find(entry.Level)->second;
97 const char* subsystem = entry.Subsystem.c_str();
98 const char* message = entry.Message.c_str();
99 const std::string timestamp = GetTimestamp(entry.Timestamp);
100
101 switch (entry.Level) {
102 case ELogLevel::Info: {
103 PrintColoredMessage(WHITE_COLOR, timestamp.c_str(), logLevel, subsystem, message);
104 } break;
105 case ELogLevel::Warning: {
106 PrintColoredMessage(YELLOW_COLOR, timestamp.c_str(), logLevel, subsystem, message);
107 } break;
108 case ELogLevel::Error: {
109 PrintColoredMessage(RED_COLOR, timestamp.c_str(), logLevel, subsystem, message);
110 } break;
111 case ELogLevel::Fatal: {
112 PrintColoredMessage(BOLD_RED_COLOR,
113 timestamp.c_str(),
114 logLevel,
115 subsystem,
116 message);
117 } break;
118 }
119 }

121 static void FormatMessage(std::string& out, const char* format, va_list args) {
122 va_list copiedArgs;
123 va_copy(copiedArgs, args);
124 const size_t bufferSize = vsnprintf(nullptr, 0, format, args);
125 const auto buffer = new char[bufferSize + 1];
126 buffer[bufferSize] = '\0';
127 vsprintf(buffer, format, copiedArgs);
128 out = STRDUP(buffer);
129 delete[] buffer;
130 }

177 void DumpToDisk() {
178 namespace fs = std::filesystem;
179 std::string logOut;
180
181 // Set log timestamp as first entry timestamp
182 time_t logTimestamp = g_LogEntries.at(0).Timestamp;
183
184 for (auto& entry : g_LogEntries) {
185 logOut += FormatEntryString(entry);
186 logOut += '\n';
187 }
188
189 const fs::path logsDir = Resources::GetRoot() / "Logs";
190 if (!exists(logsDir)) {
191 create_directory(logsDir);
192 }
193 const auto filenameFmt = fmt::format("NuggetGame_{}.log", logTimestamp);
194 const auto filename = logsDir / filenameFmt;
195
196 if (std::ofstream outputFile(filename); outputFile.is_open()) {
197 outputFile << logOut;
198 outputFile.close();
199 } else {
200 LogFatal(Subsystems::RUNTIME, "Could not write log file to disk.");
201 }
202 }
