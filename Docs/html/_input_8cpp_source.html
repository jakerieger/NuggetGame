26 void Dispatcher() {
27 while (Application::IsRunning()) {
28 for (const auto& [key, state] : g_KeyStates) {
29 // Check if the key is pressed
30 if (state.Pressed) {
31 // Dispatch OnKey event to listeners
32 FKeyEvent event {key, 0};
33 for (const auto listener : g_Listeners) {
34 listener->OnKey(event);
35 }
36 }
37 }
38 // Sleep for a short duration (120 FPS) to avoid busy waiting
39 std::this_thread::sleep_for(std::chrono::milliseconds(8));
40 }
41 }

44 void KeyCallback(GLFWwindow*, const int key, int, const int action, const int mods) {
45 const u32 _key = static_cast<u32>(key);
46 const u32 _mods = static_cast<u32>(mods);
47 FKeyEvent event {_key, _mods};
48
49 if (action == GLFW_PRESS) {
50 g_KeyStates[key].Pressed = true;
51 g_KeyStates[key].Released = false;
52 } else if (action == GLFW_RELEASE) {
53 g_KeyStates[key].Pressed = false;
54 g_KeyStates[key].Released = true;
55 }
56
57 for (const auto listener : g_Listeners) {
58 if (g_KeyStates[key].Pressed) {
59 listener->OnKeyDown(event);
60 } else if (g_KeyStates[key].Released) {
61 listener->OnKeyUp(event);
62 }
63 }
64 }

74 void MouseButtonCallback(GLFWwindow*, const int button, const int action, const int) {
75 const u32 _button = static_cast<u32>(button);
76 FMouseEvent event {_button};
77
78 for (const auto listener : g_Listeners) {
79 if (action == GLFW_PRESS && listener) {
80 listener->OnMouseDown(event);
81 } else if (action == GLFW_RELEASE && listener) {
82 listener->OnMouseUp(event);
83 }
84 }
85 }

104 void Initialize(GLFWwindow* window) {

118 void UnregisterSceneListeners(IInputListener* appListener) {
119 // Hack for .clear() not clearing pointers for some reason (I'm probably
120 // just dumb). It will leave the pointers in the array and only clear
121 // the memory associated with those pointers leading to 'Invalid

123 for (auto& listener : g_Listeners) {

127 // Remove all the active input listeners
